JavaScript — Basics (Cheat‑sheet & Examples)

Readable, copy-ready examples to learn js from first steps to advanced patterns.

1. Variables

Basic variable declarations and scope.

var oldVar = 10;      // function-scoped (avoid)
let age = 25;            // block-scoped, reassignable
const PI = 3.14;        // block-scoped, cannot be reassigned

// Example
age = 30;  // ✅ allowed
// PI = 3.14159; ❌ Error (const can't be reassigned)

const student = { name: "Name" };
student.name = "change"; // ✅ allowed (object contents can change)

console.log(oldVar, age, PI, student);

2. Data Types

Primitive and complex data types; examples and type checks.

let str = "Hello";       // String
let num = 99;             // Number
let isAdmin = true;       // Boolean
let notDefined;           // undefined
let emptyVal = null;      // null
let big = 123456789n;     // BigInt
let sym = Symbol("id");   // Symbol

console.log(typeof str, typeof num, typeof isAdmin, typeof notDefined, typeof emptyVal);

// Type coercion examples
console.log("5" + 1); // "51"  (string concatenation)
console.log("5" - 1); // 4     (number conversion)
console.log(true + 1); // 2

3. Operators

Arithmetic, assignment, comparison, logical operators and ternary.

let a = 10, b = 3;
console.log(a + b, a - b, a * b, a / b, a % b, a ** b);

let score = 5;
score += 2; // 7
console.log(score);

// Comparison
console.log(5 == "5");  // true  (loose equality)
console.log(5 === "5"); // false (strict equality)

// Logical + conditional example
let age2 = 20, hasID = true;
if (age2 >= 18 && hasID) console.log("Allowed");

// Ternary
let grade = score > 5 ? "Pass" : "Fail";
console.log(grade);

4. Control Flow

If / else and switch statements.

let marks = 78;
if (marks >= 90) {
  console.log("A");
} else if (marks >= 75) {
  console.log("B");
} else {
  console.log("C");
}

// switch
let fruit = "apple";
switch (fruit) {
  case "banana": console.log("Yellow"); break;
  case "apple": console.log("Red"); break;
  default: console.log("Unknown");
}

5. Loops

for, while, do-while, for-of, forEach and for-in.

// for loop
for (let i = 0; i < 5; i++) {
  console.log("for loop:", i);
}

// while loop
let i = 0;
while (i < 3) {
  console.log("while loop:", i);
  i++;
}

do {
  console.log("do-while loop runs at least once");
} while (false);

// for-of
for (let char of "JS") console.log("for-of:", char);

// forEach
[10, 20, 30].forEach(n => console.log("forEach:", n));

// for-in (objects)
let user = { name: "Harsh", age: 26 };
for (let key in user) console.log("for-in:", key, user[key]);

6. Functions

Function declarations, expressions, arrow functions, defaults, rest, spread, and closures.

function greet(name) {
  return "Hello " + name;
}
console.log(greet("Harsh"));

// Function expression
const add = function (x, y) { return x + y; };
console.log(add(5, 10));

// Arrow function
const multiply = (x, y) => x * y;
console.log(multiply(3, 4));

// Default parameters
function sum(a = 0, b = 0) { return a + b; }
console.log(sum(5));

// rest + spread
function total(...nums) {
  return nums.reduce((acc, val) => acc + val, 0);
}
console.log(total(1, 2, 3, 4));

let arr = [1, 2, 3];
console.log(total(...arr)); // spread

// Closure example
function counter() {
  let count = 0;
  return function () {
    count++;
    return count;
  };
}
let myCounter = counter();
console.log(myCounter(), myCounter());

7. Arrays

Common array operations: push, map, filter, reduce, find, destructuring, spread.

let fruits = ["apple", "banana", "mango"];
fruits.push("orange"); // add
console.log(fruits);

let nums = [1, 2, 3, 4, 5];
console.log(nums.map(n => n * 2));         // map
console.log(nums.filter(n => n % 2 === 0)); // filter
console.log(nums.reduce((a, b) => a + b));  // reduce
console.log(nums.find(n => n > 2));       // find

let [first, second] = fruits;
console.log(first, second);

let newNums = [...nums, 99]; // spread
console.log(newNums);

8. Objects

Object creation, property access, destructuring, copying, optional chaining, dynamic keys.

let book = {
  title: "JS Guide",
  author: "Someone",
  price: 499,
  address: { city: "Delhi", pincode: 110001 }
};
console.log(book.title, book["author"]);
console.log(book.address.city);

// Destructuring
let { title, price } = book;
console.log(title, price);

// Loop through object
for (let key in book) console.log(key, book[key]);

console.log(Object.keys(book));
console.log(Object.entries(book));

// Copy objects
let shallowCopy = { ...book };
let deepCopy = JSON.parse(JSON.stringify(book));

console.log(book?.publisher?.name); // optional chaining example
console.log(deepCopy);

let keyName = "rating";
let review = { [keyName]: 4.5 };
console.log(review);
